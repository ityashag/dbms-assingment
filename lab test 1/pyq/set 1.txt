create table toys (toys_nums int primary key ,name varchar(20), price int , age_group varchar(10) );

create table manufacturer(manu_id int primary key,name varchar(20), toys_nums int primary key, foreign key(toys_nums) references toys(toys_nums));

create table customer(cust_id int,name varchar(20),city varchar(20), last_order_data date, toys_nums int, 
primary key(cust_id,toys_nums), foreign key(toys_nums) references toys(toys_nums));
